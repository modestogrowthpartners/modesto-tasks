-- ===================================================================== -- Modesto — habilitar os tipos de documento novos (diagnostico e weekly) -- e permitir cadastrar documento por link. -- -- Este script DESCOBRE sozinho como a coluna `tipo` foi criada no seu -- banco (enum com qualquer nome, ou texto com restrição) e se adapta. -- Rode tudo de uma vez no Supabase → SQL Editor. -- ===================================================================== -- --------------------------------------------------------------------- -- PARTE 1 — diagnóstico (só mostra, não altera nada) -- --------------------------------------------------------------------- select column_name as coluna, data_type as tipo_da_coluna, udt_name as nome_interno, is_nullable as aceita_vazio from information_schema.columns where table_schema = 'public' and table_name = 'documents' order by ordinal_position; -- restrições da tabela (se o tipo for texto com CHECK, aparece aqui) select conname as restricao, pg_get_constraintdef(oid) as definicao from pg_constraint where conrelid = 'public.documents'::regclass; -- --------------------------------------------------------------------- -- PARTE 2 — aplica a mudança, seja qual for o formato -- --------------------------------------------------------------------- do $$ declare v_enum text; -- nome do enum, se for enum v_check text; -- nome da restrição CHECK, se for texto v_novos text[] := array['diagnostico','weekly']; v_valor text; begin -- descobre se `tipo` é um enum e qual o nome dele select t.typname into v_enum from pg_attribute a join pg_type t on t.oid = a.atttypid where a.attrelid = 'public.documents'::regclass and a.attname = 'tipo' and t.typtype = 'e'; -- 'e' = enum if v_enum is not null then -- ---- caso A: é enum → acrescenta os valores que faltam ---- foreach v_valor in array v_novos loop if not exists ( select 1 from pg_enum e join pg_type t on t.oid = e.enumtypid where t.typname = v_enum and e.enumlabel = v_valor ) then execute format('alter type %I add value %L', v_enum, v_valor); raise notice 'Valor "%" acrescentado ao enum %', v_valor, v_enum; else raise notice 'Valor "%" já existia em %', v_valor, v_enum; end if; end loop; else -- ---- caso B: é texto com CHECK → recria a restrição incluindo os novos ---- select conname into v_check from pg_constraint where conrelid = 'public.documents'::regclass and contype = 'c' and pg_get_constraintdef(oid) ilike '%tipo%'; if v_check is not null then execute format('alter table public.documents drop constraint %I', v_check); raise notice 'Restrição % removida', v_check; end if; alter table public.documents add constraint documents_tipo_check check (tipo in ('apresentacao','concorrencia','proposta','diagnostico','weekly')); raise notice 'Restrição recriada com os tipos novos'; end if; end $$; -- --------------------------------------------------------------------- -- PARTE 3 — permitir documento cadastrado por link (sem subir arquivo) -- Quem sobe arquivo pelo botão "Enviar documento" continua igual. -- --------------------------------------------------------------------- alter table public.documents alter column storage_path drop not null; -- --------------------------------------------------------------------- -- PARTE 4 — conferência: mostra os tipos aceitos agora -- --------------------------------------------------------------------- select e.enumlabel as tipos_aceitos from pg_attribute a join pg_type t on t.oid = a.atttypid join pg_enum e on e.enumtypid = t.oid where a.attrelid = 'public.documents'::regclass and a.attname = 'tipo' union all select unnest(array['(coluna é texto — veja a restrição CHECK acima)']) where not exists ( select 1 from pg_attribute a join pg_type t on t.oid = a.atttypid where a.attrelid = 'public.documents'::regclass and a.attname = 'tipo' and t.typtype = 'e' );
